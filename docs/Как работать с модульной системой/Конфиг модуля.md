КОНФИГ МОДУЛЯ:

--------------------------------------------
В файле `modules/<имя_модуля>/<имя_модуля>_config.py` мы создаём конфигурацию для нашего модуля.

```python
# config это базовые классы для создания конфигурации самого модуля и его роутов
from app.config.module_config import ModuleConfig
from app.config.router_config import RouterConfig

# types 
from app.types import EMethod, TRouter
from typing import Dict

# Класс для работы с путями в модуле 
class <имя_модуля>RouterConfig(RouterConfig):
    def __init__(self, routers: Dict[str, TRouter]):
        super(<имя_модуля>RouterConfig, self).__init__(routers)

# Основной конфигурационный класс всего модуля
class <имя_модуля>ModuleConfig(ModuleConfig):
    def __init__(self, 
        title: str, 
        description: str, 
        r: <имя_модуля>RouterConfig,
        is_logging: bool = True 
    ) -> None:
        super(<имя_модуля>ModuleConfig, self).__init__(
            title=title,
            description=description,
            r=r,
            is_logging=is_logging
        )


# Дальше идёт hashMap роутов для модуля
routers = {
    "Название роута(например, 'Добавление нового админа')": TRouter(
        path = "/path/to/route", # путь по которому будет доступен роут
        methods = [EMethod.GET, EMethod.POST], # методы, которые будут доступны для этого роута
        template = "admin/add_admin_page.html" # путь к шаблону, который будет использоваться для этого роута
    ),
    ... # другие роуты
}


# Создаём конфигурацию для модуля
<имя_модуля>Conf = <имя_модуля>ModuleConfig(
    title = "Модуль: app.modules.<имя_модуля>", # название модуля и можно указать путь к модулю, чтобы улучшить логирование
    description = """
    Модуль для работы с логикой <имя_модуля>. 
    Основные функции:
    - ...
    - ...
    ...

    Тесты модуля:
    python3.12 -m unittest app/modules/<имя_модуля>/<имя_модуля>_test.py
    """,
    r = <имя_модуля>RouterConfig(routers), # конфигурация роутов для модуля
    is_logging = True # логирование для модуля
)
```

Отлично, мы создали конфигурацию для модуля.
